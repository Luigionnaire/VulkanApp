51 if (result == VK_ERROR_OUT_OF_DATE_KHR) { // check if the swapchain is out of date

71 submitInfo.waitSemaphoreCount = 1; // number of semaphores to wait for

73 submitInfo.pWaitDstStageMask = waitStages; // stages to wait for

80 if (vkQueueSubmit(m_device->getGraphicsQueue(), 1, &submitInfo, inFlightFences[currentFrame]) != VK_SUCCESS) { // submit the command buffer !!memory access!!

135void Renderer::createSyncObjects() {
136 imageAvailableSemaphores.resize(MAX_FRAMES_IN_FLIGHT);
137 renderFinishedSemaphores.resize(MAX_FRAMES_IN_FLIGHT);
138 inFlightFences.resize(MAX_FRAMES_IN_FLIGHT);
139
140 VkSemaphoreCreateInfo semaphoreInfo{};
141 semaphoreInfo.sType = VK_STRUCTURE_TYPE_SEMAPHORE_CREATE_INFO;
142
143 VkFenceCreateInfo fenceInfo{};
144 fenceInfo.sType = VK_STRUCTURE_TYPE_FENCE_CREATE_INFO;
145 fenceInfo.flags = VK_FENCE_CREATE_SIGNALED_BIT; // signaled so it doesnt hang at the first frame
146
147 for (size_t i = 0; i < MAX_FRAMES_IN_FLIGHT; i++) {
148 if (vkCreateSemaphore(m_device->getDevice(), &semaphoreInfo, nullptr, &imageAvailableSemaphores[i]) != VK_SUCCESS
149 || vkCreateSemaphore(m_device->getDevice(), &semaphoreInfo, nullptr, &renderFinishedSemaphores[i]) != VK_SUCCESS
150 || vkCreateFence(m_device->getDevice(), &fenceInfo, nullptr, &inFlightFences[i]) != VK_SUCCESS) // create semaphores and fence

157void Renderer::recordCommandBuffer(VkCommandBuffer commandBuffer, uint32_t imageIndex) {
158 VkCommandBufferBeginInfo beginInfo{};
159 beginInfo.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
160 beginInfo.flags = 0; // optional
161 beginInfo.pInheritanceInfo = nullptr; // optional
162
163 if (vkBeginCommandBuffer(commandBuffer, &beginInfo) != VK_SUCCESS) {
164 throw std::runtime_error("failed to begin recording command buffer!");
165 }

167 VkExtent2D swapChainExtent = m_swapChain->getSwapChainExtent();
168 //RENDER PASS
169 VkRenderPassBeginInfo renderPassInfo{};
170 renderPassInfo.sType = VK_STRUCTURE_TYPE_RENDER_PASS_BEGIN_INFO;
171 renderPassInfo.renderPass = m_renderPass->getRenderPass();
172 renderPassInfo.framebuffer = m_swapChain->getSwapChainFramebuffers()[imageIndex];
173 renderPassInfo.renderArea.offset = { 0, 0 };
174 renderPassInfo.renderArea.extent = swapChainExtent;
175
176 std::array<VkClearValue, 2> clearValues{}; // clear color
177 clearValues[0].color = { 0.012f, 0.018f, 0.02f, 1.0f }; // clear color
178 clearValues[1].depthStencil = { 1.0f, 0 }; // clear depth
179
180 renderPassInfo.clearValueCount = static_cast<uint32_t>(clearValues.size());
181 renderPassInfo.pClearValues = clearValues.data(); // clear color
182
183 vkCmdBeginRenderPass(commandBuffer, &renderPassInfo, VK_SUBPASS_CONTENTS_INLINE); // begin render pass
184
185 vkCmdBindPipeline(commandBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS, m_pipeline->getPipeline()); // bind the pipeline
186
187 //VIEWPORT AND SCISSOR
188 VkViewport viewport{};
189 viewport.x = 0.0f;
190 viewport.y = 0.0f;
191 viewport.width = static_cast<float>(swapChainExtent.width);
192 viewport.height = static_cast<float>(swapChainExtent.height);
193 viewport.minDepth = 0.0f;
194 viewport.maxDepth = 1.0f;
195 vkCmdSetViewport(commandBuffer, 0, 1, &viewport); // set the viewport
196
197 VkRect2D scissor{};
198 scissor.offset = { 0, 0 };
199 scissor.extent = swapChainExtent;
200 vkCmdSetScissor(commandBuffer, 0, 1, &scissor); // set the scissor
201
203 m_modelPBR->bind(commandBuffer); // bind the buffers
204
205 //DESCRIPTOR SETS
206 vkCmdBindDescriptorSets(commandBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS, m_pipeline->getPipelineLayout(), 0, 1, &m_descriptorManager->getDescriptorSet(currentFrame), 0, nullptr); // bind the descriptor sets
207
208 m_modelPBR->draw(commandBuffer); // draw the triangle
209

211 if (vkEndCommandBuffer(commandBuffer) != VK_SUCCESS) {
