57void VKSwapChain::createSwapChain(VkSurfaceKHR surface) {
58 SwapChain::SwapChainSupportDetails swapChainSupport = SwapChain::querySwapChainSupport(m_physicalDevice, surface);
59
60 VkSurfaceFormatKHR surfaceFormat = chooseSwapSurfaceFormat(swapChainSupport.formats);
61 VkPresentModeKHR presentMode = chooseSwapPresentMode(swapChainSupport.presentModes);
62 VkExtent2D extent = chooseSwapExtent(swapChainSupport.capabilities, m_window);
63
64 uint32_t imageCount = swapChainSupport.capabilities.minImageCount + 1;
65 if (swapChainSupport.capabilities.maxImageCount > 0 && imageCount > swapChainSupport.capabilities.maxImageCount) {
66 imageCount = swapChainSupport.capabilities.maxImageCount;
67 }
68
69 VkSwapchainCreateInfoKHR createInfo{};
70 createInfo.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
71 createInfo.surface = surface;
72 createInfo.minImageCount = imageCount;
73 createInfo.imageFormat = surfaceFormat.format;
74 createInfo.imageColorSpace = surfaceFormat.colorSpace;
75 createInfo.imageExtent = extent;
76 createInfo.imageArrayLayers = 1;
77 createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
78
79 QueueFamily::QueueFamilyIndices indices = QueueFamily::findQueueFamilies(m_physicalDevice, surface);
80 uint32_t queueFamilyIndices[] = { indices.graphicsFamily.value(), indices.presentFamily.value() };
81 if (indices.graphicsFamily != indices.presentFamily) { // doing this for ownership reasons (lookup)
82 createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT; // concurrent mode
83 createInfo.queueFamilyIndexCount = 2;
84 createInfo.pQueueFamilyIndices = queueFamilyIndices; // set the indices
85 }
86 else {
87 createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE; // exclusive mode
88 createInfo.queueFamilyIndexCount = 0; // optional
89 createInfo.pQueueFamilyIndices = nullptr; // optional
90 }
91 createInfo.preTransform = swapChainSupport.capabilities.currentTransform;
92 createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
93 createInfo.presentMode = presentMode;
94 createInfo.clipped = VK_TRUE; // ignore pixels that are obscured(like if behind another window)
95 createInfo.oldSwapchain = VK_NULL_HANDLE; // resizing stuff, look up later
96
97 if (vkCreateSwapchainKHR(m_device, &createInfo, nullptr, &m_swapChain) != VK_SUCCESS) {
98 throw std::runtime_error("failed to create swap chain!");
99 }
100 vkGetSwapchainImagesKHR(m_device, m_swapChain, &imageCount, nullptr);
101 m_swapChainImages.resize(imageCount);
102 vkGetSwapchainImagesKHR(m_device, m_swapChain, &imageCount, m_swapChainImages.data());
103
104 // save these to member variables for later
105 m_swapChainImageFormat = surfaceFormat.format;
106 m_swapChainExtent = extent;

168VkExtent2D VKSwapChain::chooseSwapExtent(const VkSurfaceCapabilitiesKHR& capabilities, GLFWwindow* window) {
169 if (capabilities.currentExtent.width != std::numeric_limits<uint32_t>::max()) {
170 return capabilities.currentExtent;
171 }
172 else {
173 int width, height;
174 glfwGetFramebufferSize(window, &width, &height);

187VkFormat VKSwapChain::findSupportedFormat(const std::vector<VkFormat>& candidates, VkImageTiling tiling, VkFormatFeatureFlags features)
188{
189 for (VkFormat format : candidates) {
190 VkFormatProperties props;
191 vkGetPhysicalDeviceFormatProperties(m_physicalDevice, format, &props);
192 if (tiling == VK_IMAGE_TILING_LINEAR && (props.linearTilingFeatures & features) == features) {
193 return format;
194 }
195 else if (tiling == VK_IMAGE_TILING_OPTIMAL && (props.optimalTilingFeatures & features) == features) {
196 return format;
197 }
198 }
199 throw std::runtime_error("failed to find supported format!");
200}
